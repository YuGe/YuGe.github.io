我的vimrc文件

blog | Jul 12, 2014

这里贴出了我的vimrc文件，一些设置后面会有注释，但是没有特别详细的说明。

这个配置文件能够通用与Mac和目前流行的发行版Linux，在Windows下能够绝大部分通用，但是可能会需要根据具体情况进行一下小的修改。

我使用的是Vundle来管理插件，使用之前需要下载安装一下Vundle，然后Bundle Install便可以安装所有的插件。

YouCompleteMe是一个非常优秀的代码补全插件，且提供了多种语言的语义补全，但是安装的时候需要编译，有一点复杂。

"""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""
"                                preamble
"""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""

set nocompatible

"""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""
"                                 Pulgins
"""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""

Plug 'Yggdroot/indentLine', {'on': 'IndentLinesToggle'}
Plug 'rizzatti/dash.vim', {'on': 'DashSearch'}

Plug 'sjl/gundo.vim', {'on': 'GundoToggle'}
Plug 'yegappan/mru'
Plug 'terryma/vim-multiple-cursors'
Plug 'vim-scripts/YankRing.vim'
Plug 'jlanzarotta/bufexplorer'
Plug 'junegunn/vim-easy-align'
Plug 'Raimondi/delimitMate'
Plug 'Valloric/ListToggle'
Plug 'tomtom/tcomment_vim'
Plug 'tpope/vim-surround'
Plug 'tpope/vim-repeat'
Plug 'kien/ctrlp.vim'
Plug 'vim-scripts/matchit.zip', {'for': ['html', 'htmldjango']}

"""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""
"                        turn on filetype plugins
"""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""
" Enable detection, plugins and indenting
filetype plugin indent on

"""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""
"                            General settings
"""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""

syntax enable                       " Enable syntax highlighting
let g:solarized_termcolors=256
colorscheme solarized
set background=light
set autoindent
set smartindent
set smarttab                        " Be smart when using tabs
set autoread                        " Auto read when file changed outside
set autowrite                       " Auto write
set backspace=indent,eol,start      " Configure backspace
" set clipboard+=unnamedplus,unnamed  " Share clipboard with system
set completeopt=menu,menuone	    " Close preview window
set hlsearch                        " Highlight search results
set incsearch                       " Show result while type
set ignorecase                      " Ignore case when searching
set smartcase                       " Smart about cases when searching
set history=1000                    " History number
set lazyredraw                      " Don't redraw while executing macros
set hidden                          " Buffer becomes hidden when abandoned
set magic                           " For regular expressions turn magic on
set showmatch                       " Show matching brackets
set sidescrolloff=6                 " Horizontal scrolloff
set scrolloff=6                     " 6 lines above/below cursor when scroll
set textwidth=80                    " Set the length of a line
set colorcolumn=+1                  " Highlight column after text width
set vb t_vb=                        " No beep or flash warning
set mouse=a                         " Enable mouse
set linebreak                       " Wrap lines at a character in 'breakat'
set mat=1                           " How many tenths of seconds to blink
set title                           " Show file in title bar
set cmdheight=2                     " Height of the command bar
set showcmd                         " Show partial command in the last line
set showmode                        " Show current mode in the last line
set wildmenu                        " Completion with menu
set nowrap                          " Don't wrap lines
set noswapfile                      " No swap file
set backup                          " Turn backup on
set backupdir=$HOME/.vim/backup     " Set backup files directory
set undofile                        " Turn persistent undo on
set undodir=$HOME/.vim/undo         " Set backup files directory
set undolevels=800
set undoreload=800
set viminfo^='1000,<500             " Remember info about open buffers
set wildignore=*.so,*.o,*.pyc,*~    " Ignored files
set fileformat=unix                 " Default file format
set fileformats=unix,dos,mac        " Detects file formats in order
set encoding=utf-8                  " Character encoding used inside VIM
set fileencodings=utf-8,ucs-bom,gb18030 " Detects file encodings in order

" Status line
set laststatus=2                 		" Always have a status line
set statusline =%#Special#\ %n\ %#              " buffer number
set statusline +=Directory#%{&ff}%#             " file format
set statusline +=Statement#%y%#                 " file type
set statusline +=Constant#\ %<%-0.30F%#         " full path
set statusline +=Special#%m\ %#                 " modified flag
set statusline +=Directory#%=%#                 " separation
set statusline +=Statement#\ %l%#               " current line
set statusline +=Directory#/%L%#                " total lines
set statusline +=Statement#\ %v%#               " virtual column number
set statusline +=Directory#\ %p%%\ %#           " percentage of the file

" Set extra options when running in GUI mode
if has("GUI_running")
    set lines=36
    set columns=85
    set guioptions-=rL
    set relativenumber
    set paste                  "set paste in terminal will influence <tab>
endif

" Set font according to system
if has("mac")
    set guifont=Source\ code\ pro:h12
end

" Source vimrc when saving
autocmd! BufWritePost $MYVIMRC source $MYVIMRC

" Highlight Trailing whitespace
highlight TrailSpace ctermbg=red guibg=red
match TrailSpace /\s\+$/

" Delete trailing DOS-returns, tab, and whitespace
autocmd BufRead,BufWritePre,FileWritePre * silent! %s/[\r \t]\+$//

" Set filetype for somefile
auto BufNewFile,BufReadPost *.h set filetype=c
auto BufNewFile,BufReadPost *.pbs set filetype=sh

""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""
"                             custom mappings
""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""

let g:mapleader = ","

" Stop highlighting
nnoremap <leader><cr> :nohlsearch<cr>

" Fast editing vimrc
map <silent> <leader>ee :e! $MYVIMRC<cr>

" Switch to the directory of the open buffer
noremap <leader>cd :cd %:p:h<cr>

" Close the current buffer
map <leader>bd :bd!<cr>

" Toggle and spell checking
noremap <leader>ss :setlocal spell! spelllang=en_us<cr>

" Keep search matches in the middle of the window.
" zz centers the screen on the cursor, zv unfolds any fold if the cursor
" suddenly appears inside a fold.
nnoremap * *zzzv
nnoremap # #zzzv
nnoremap n nzzzv
nnoremap N Nzzzv

" key bindings for quickly moving between windows
noremap <c-h> <c-w>h
noremap <c-l> <c-w>l
noremap <c-k> <c-w>k
noremap <c-j> <c-w>j

" Scrolling leave the line at the center of the window
noremap <c-u> <c-u>zz
noremap <c-d> <c-d>zz
noremap <c-f> <c-f>zz
noremap <c-b> <c-b>zz

" Easier moving of code blocks
vnoremap < <gv
vnoremap > >gv

" Bash like keys for the command line
cnoremap <C-A>  <Home>
cnoremap <C-E>  <End>

"""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""
"                          PLUGIN CONFIGURATION
"""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""

" Ctrlp
let g:ctrlp_max_height=15
let g:ctrlp_working_path_mode=0
let g:ctrlp_map = '<leader>,'

" Dash
nmap <silent> <leader>ds <Plug>DashSearch

" DelimitMate
let delimitMate_nesting_quotes = ['"', '`']
let delimitMate_expand_cr = 1
let delimitMate_expand_space = 1

" Easy align
xmap ga <plug>(EasyAlign)
nmap ga <plug>(EasyAlign)

" Gundo
nnoremap <F8> :GundoToggle<cr>
let g:gundo_width=30

" IndentGuide
let g:indentLine_enabled = 0
let indentLine_char = '│'
nmap <silent> <Leader>ii :IndentLinesToggle<cr>
nmap <silent> <Leader>ie :IndentLinesEnable<cr>
nmap <silent> <Leader>id :IndentLinesDisable<cr>

" InstantMarkdown
let g:instant_markdown_autostart = 0
let g:instant_markdown_slow = 1

" ListToggle
let g:lt_height = 10
let g:lt_location_list_toggle_map = '<leader>l'
let g:lt_quickfix_list_toggle_map = '<leader>q'

" MRU
let MRU_Entries = 80
let MRU_File = $HOME . '/.vim/tmp/vim_mru_files'
map <leader>r :MRU<cr>

" Nerdtree
let g:NERDTreeWinSize=25
let g:NERDTreeMinimalUI=1
let g:NERDTreeIgnore=['\.vim$', '\~$','\.pyc$']
map <leader>nn :NERDTreeToggle<cr>
map <leader>nc :NERDTreeCWD<cr>

" Tagbar
let g:tagbar_width= 25
let g:tagbar_autofocus = 1
let g:tagbar_compact = 1
let g:tagbar_show_visibility = 0
let g:tagbar_indent = 1
let g:tagbar_foldlevel = 2
let g:tagbar_ctags_bin = '/usr/local/bin/ctags'
map <leader>tt :TagbarToggle<cr>

" Yankring
let g:yankring_min_element_length = 2
let g:yankring_history_dir = $HOME . '/.vim/tmp/'
map <silent> <leader>y :YRShow<cr>

" Syntastic
let g:syntastic_always_populate_loc_list = 1
let g:syntastic_auto_loc_list = 1
let g:syntastic_check_on_open = 0
let g:syntastic_check_on_wq = 0
let g:syntastic_javascript_checkers = ['jshint']
let g:syntastic_python_checkers=['flake8']
let g:syntastic_python_flake8_args='--ignore=E501'
let g:syntastic_html_tidy_exec = 'tidy5'

" YouCompleteMe
let g:ycm_confirm_extra_conf = 0
let g:ycm_complete_in_comments = 1
let g:ycm_collect_identifiers_from_comments_and_strings = 1
let g:ycm_key_list_select_completion = ['<C-N>']
let g:ycm_key_list_previous_completion = ['<C-P>']
nnoremap <leader>gl :YcmCompleter GoToDeclaration<CR>
nnoremap <leader>gf :YcmCompleter GoToDefinition<CR>
nnoremap <leader>gg :YcmCompleter GoToDefinitionElseDeclaration<CR>
